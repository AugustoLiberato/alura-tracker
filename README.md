# alura-tracker

# 1. Clonar o repositório
git clone [url-do-repositorio]

# 2. Entrar na pasta
cd alura-tracker

# 3. Instalar todas as dependências (incluindo json-server)
npm install

# 4. Rodar o backend
npm run backend

# 5. Em outro terminal, rodar o frontend
npm run serve
